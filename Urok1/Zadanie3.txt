
<?php
$a = 5;
$b = '05';
var_dump($a == $b);                             // Почему true? - происходит сравнение по значению [5=5]
var_dump((int)'012345');                        // Почему 12345? - т.к. переменной задан тип integer, который представляет целое число
var_dump((float)123.0 === (int)123.0); // Почему false? - происходит сравнение по значению и типу [типы разные]
var_dump((int)0 === (int)'hello, world'); // Почему true? - т.к. оператор тождественного равенства сначала сравнивает типы переменных (в данном случае они одинаковые), а затем сравнивает значения переменных. Но любая строка не содержащая вв начале чисел равнется нулю при пееводе в число, следовательно, значения переменных равны
?>
